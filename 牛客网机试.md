```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <cstdio>
#include <climits>
#include <string>
#include <vector>
#include <stack>
#include <map>
#include <set>
#include <cmath>
#include <algorithm>
using namespace std;
const int maxn = 6;
// 此题用深搜
int mymap[maxn][maxn];
int dir[4][2] = { {0, -1}, {0, 1}, {-1, 0}, {1, 0} };
int minCost = INT_MAX;
int startx, starty, endx, endy;

bool isValid(int x, int y) {
	return (x >= 0 && x <= 5) && (y >= 0 && y <= 5);
}

void dfs(int sx, int sy, int nowCost, int nowState, bool visited[6][6]) {
	if (sx == endx && sy == endy) {
		if (nowCost < minCost) {
			minCost = nowCost;
		}
		return;
	}
	visited[sx][sy] = true;
	for (int k = 0; k < 4; k++) {
		int x = sx + dir[k][0];
		int y = sy + dir[k][1];
		if (!isValid(x, y) || visited[x][y])
			continue;
		int cost = nowState * mymap[x][y];
		int tmpnowCost = nowCost + cost;
		int tmpnowState = cost % 4 + 1;
		dfs(x, y, tmpnowCost, tmpnowState, visited);
	}
	visited[sx][sy] = false;
}

int main() {
	bool visited[maxn][maxn];
	for (int i = 0; i < maxn; i++) {
		for (int j = 0; j < maxn; j++) {
			cin >> mymap[i][j];
			visited[i][j] = false;
		}
	}
	cin >> startx >> starty >> endx >> endy;
	dfs(startx, starty, 0, 1, visited);
	cout << minCost << endl;
	system("pause");
	return 0;
}
```

```c++
// 分治思想，快速找到第K大元素
// 左边的小，右边的大，中间的相等
int findNumberK(vector<int> &vec, int k) {
	vector<int> s_vec, e_vec, l_vec;
	int pivot = vec[vec.size() / 2];        // 选取参考值
	for (int i = 0; i < vec.size(); i++) {  // 遍历并分组
		if (vec[i] < pivot) {
			s_vec.push_back(vec[i]);
		}
		else if (vec[i] == pivot) {
			e_vec.push_back(vec[i]);
		}
		else {
			l_vec.push_back(vec[i]);
		}
	}
	if (k <= s_vec.size()) {                // 进入下一层递归
		return findNumberK(s_vec, k);
	}
	else if (k > s_vec.size() + e_vec.size()) {
		return findNumberK(l_vec, k - s_vec.size() - e_vec.size());
	}
	else {
		return pivot;
	}
}
```

```c++
class BigNum {
	int a[1000];
	const int MAX = 10;  // 十进制，可以换为2进制呀8进制
	const int size = 1000;
public:
	BigNum() { fill(a, a + size, 0); }
	BigNum(string s) {  // !!!倒着存储
		fill(a, a + size, 0);
		for (int j = 0, i = s.size() - 1; i >= 0; i--, j++) {
			a[j] = s[i] - '0';
		}
	}
	void operator+(BigNum& b) {  // !!!这个运算符重载
		BigNum c;
		int carry = 0;
		for (int i = 0; i < size; i++) {
			c.a[i] = a[i] + b.a[i] + carry;
			carry = c.a[i] / MAX;  
			c.a[i] = c.a[i] % MAX;
		}
		c.print();
	}

	void operator-(BigNum& b) {
		int pos = size - 1;
		while (a[pos] == b.a[pos]) pos--;
		if (a[pos] < b.a[pos]) {  // !!!小数减大数，输出负号，转成大数减小数
			for (int i = 0; i < size; i++) {
				swap(a[i], b.a[i]);
			}
			cout << '-';
		}

		BigNum c;
		int carry = 0;
		for (int i = 0; i < size; i++) {
			c.a[i] = a[i] - b.a[i] - carry;
			if (c.a[i] < 0) {  // !!!向高位借位
				carry = 1;
				c.a[i] = c.a[i] + MAX;
			}
			else {
				carry = 0;
				c.a[i] = c.a[i];
			}
		}
		c.print();
	}

	void operator*(BigNum& b) {  // !!!每次计算的是c[i+j]这一位!!!
		BigNum c;
		int carry = 0;
		for (int i = 0; i < 500; i++) {  
			int carry = 0;
			for (int j = 0; j < 500; j++) {
				c.a[i + j] = a[i] * b.a[j] + c.a[i + j] + carry;
				carry = c.a[i + j] / MAX; 
				// !!!这里很巧妙，因为正好往后面几位遍历，自然处理carry每一位
				c.a[i + j] = c.a[i + j] % MAX;
			}
		}
		c.print();
	}

	void print() {
		int i = size - 1;
		while (a[i] == 0) i--;
		for (; i >= 0; i--) {
			cout << a[i];
		}
		cout << endl;
	}
};

		a * b;
	}
```

```c++
void dfs(int sta) {
	visited[sta] = true;
	for (int i = 0; i < graph[sta].size(); i++) {
		if (!visited[graph[sta][i]]) {
			dfs(graph[sta][i]);
		}
	}
}

	fill(visited, visited + maxn, true);
	int i, j;
	int count = 0;

		graph[i].push_back(j);
		graph[j].push_back(i);

		if (!visited[k]) {

		}
```

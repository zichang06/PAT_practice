## 日期处理 P91
```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
using namespace std;

int month[13][2] = { {0,0}, {31,31}, {28, 29}, {31,31}, {30, 30}, {31,31}, 
{30, 30}, {31,31}, {31, 31}, {30, 30}, {31,31}, {30, 30}, {31,31} };

bool isLeap(int year) {
	return((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0));
}

int main() {
	int time1, y1, m1, d1;
	int time2, y2, m2, d2;
	while (scanf("%d%d", &time1, &time2) != EOF) {
		if (time1 > time2) {
			int temp = time1;
			time1 = time2;
			time2 = temp;
		}

		y1 = time1 / 10000;
		m1 = time1 % 10000 / 100;
		d1 = time1 % 100;
		y2 = time2 / 10000;
		m2 = time2 % 10000 / 100;
		d2 = time2 % 100;

		int ans = 1;
		while (y1 < y2 || m1 < m2 || d1 < d2) {
			d1++;
			if (d1 == month[m1][isLeap(y1)] + 1) {
				m1++;
				d1 = 1;
			}
			if (m1 == 13) {
				y1++;
				m1 = 1;
			}
			ans++;
		}
		printf("%d\n", ans);
	}
	system("pause");
	return 0;
}
```
## 进制转换 P93
1. 将P进制数x转换为十进制数y
```c++
int P2ten(int p, int x) {
	int y = 0;
	int product = 1;
	while (x != 0) {
		y += (x % 10) * product;
		x /= 10;
		product *= p;
	}
	return y;
}
```
2. 将十进制数y转换为Q进制数z —— 除基取余法
```c++
int z[40];  // 返回的数组要反过来输出
int ten2Q(int q, int y, int* z) {
	int num = 0;
	do {
		z[num++] = y % q;
		y /= q;
	} while (y != 0);
	return num; // 返回位数
}
```
